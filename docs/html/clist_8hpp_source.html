38// CList objects replace Prolog lists for efficiency
39// WARNING: Because the internal list is mutable care must be taken when using
40// CList objects. For example, it might be necessary to copy the list before
41// passing it to a function that might modify it or when binding a variable to
42// a CList object.
43// If a Prolog-like list is required, a user-defined class that implements the
44// Term interface should be used (for example ConsList).
45

56class CList : public Term {
57public:
61 CList() {}
62
67 CList(const std::list<Term *> &elements) : elements(elements) {}
68
73 Term *dereference() override { return this; }
74
80 bool bind(Term *t) override {
81 return false; // Lists cannot be bound to other terms
82 }
83
88 void reset(Term *t) override {
89 // No-op for CList
90 }
91
96 std::string repr() const override {
97 std::ostringstream oss;
98 oss << "[";
99 for (auto it = elements.begin(); it != elements.end(); ++it) {
100 if (it != elements.begin()) {
101 oss << ", ";
102 }
103 oss << (*it)->repr();
104 }
105 oss << "]";
106 return oss.str();
107 }
108
114 bool isEqualTo(Term &t) override {
115 CList *l = dynamic_cast<CList *>(&t);
116 if (l == nullptr)
117 return false;
118 return elements == l->elements;
119 }
120
126 bool isLessThan(Term &t) override;
127
132 void addElement(Term *element) { elements.push_back(element); }
133
138 const std::list<Term *> &getElements() const { return elements; }
139
140private:
141 std::list<Term *> elements;
142};
