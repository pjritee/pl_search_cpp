40// CList objects replace Prolog lists for efficiency
41// The list passed to the constructor is copied, so it can be safely modified
42// after the CList is created. If a Prolog-like list is required, a user-defined
43// class that implements the Term interface should be used (for example
44// ConsList).
45
46namespace pl_search {
47
56class CList : public Term {
57protected:
58 std::list<TermPtr> elements;
59
60public:
65 CList(std::list<TermPtr> elems) : elements(elems) {}
66
71 std::string repr() const override {
72 std::ostringstream oss;
73 oss << "[";
74 for (auto it = elements.begin(); it != elements.end(); ++it) {
75 if (it != elements.begin()) {
76 oss << ", ";
77 }
78 oss << (*it)->repr();
79 }
80 oss << "]";
81 return oss.str();
82 }
83
89 bool isEqualTo(Term &other) const override {
90 if (CList *list = dynamic_cast<CList *>(&other)) {
91 return elements == list->elements;
92 }
93 return false;
94 }
95
101 bool isLessThan(Term &other) const override;
102
108 void addElement(TermPtr element) { elements.push_back(element); }
109
114 const std::list<TermPtr> &getElements() const { return elements; }
115};
