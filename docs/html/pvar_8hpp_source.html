105// UpdatableVar is used to implement what some Prologs call
106// updatable assignment. This is typically used to store (part of) the
107// state in a way that can be backtracked over. For example, after
108// binding a variable and then making some deductions the available choices
109// for another variable might have been reduced and we can use an UpdatableVar
110// to store that value as the computation moves forward but on backtracking
111// the old value will be restored
112class UpdatablePVar : public PVar {
113public:
114 UpdatablePVar(Term *t) : PVar() { value = t; }
115
116 // As the intention is to update the value of the variable, we need to
117 // simply return this pointer so that a following bind will update the value.
118 // If we wanted to do a full dereference for an updatable variable, v, we
119 // would use v->PVar::dereference() or v->getValue()->dereference()
120 Term *dereference() override { return this; }
121};
