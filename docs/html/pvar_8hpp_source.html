114/*
115 * This is typically used to store (part of) the
116 * state in a way that can be backtracked over. For example, after
117 * binding a variable and then making some deductions the available choices
118 * for another variable might have been reduced and we can use an UpdatableVar
119 * to store that value as the computation moves forward but on backtracking
120 * the old value will be restored.
121 */
122class UpdatablePVar : public PVar {
123
124public:
125 UpdatablePVar(TermPtr t) : PVar() { value = t; }
126
127 // As the intention is to update the value of the variable, we need to
128 // simply return this pointer so that a following bind will update the value.
129 // If we wanted to do a full dereference for an updatable variable, v, we
130 // would use v->PVar::dereference() or v->getValue()->dereference()
131 TermPtr dereference() override { return shared_from_this(); }
132
133protected:
134 Term *deref_term() override { return this; }
135};
