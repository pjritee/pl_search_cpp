195class DisjPred : public Pred {
196public:
205 DisjPred(Engine *eng, std::vector<PredPtr> preds)
206 : Pred(eng), preds(preds), index(0) {}
207
208 void initialize_call() override {
210 for (auto it = preds.begin(); it != preds.end(); it++) {
211 (*it)->set_continuation(continuation);
212 }
213 };
214
215 bool apply_choice() override {
217 PredPtr choice = preds[index++];
222 continuation = choice;
223 return true;
224 }
225
226 bool more_choices() override { return index < preds.size(); }
227
228private:
229 std::vector<PredPtr> preds;
230 int index;
231};
